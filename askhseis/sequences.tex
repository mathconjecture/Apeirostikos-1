\section{Φραγμένες Ακολουθίες}

\begin{enumerate}
    \item Να δείξετε ότι η ακολουθία $ a_{n} = \frac{n}{3^{n}} $ είναι φραγμένη. 
        \hfill Απ: $ \abs{a_{n}}< \frac{1}{2} $
    \item Να δείξετε ότι η ακολουθία $ a_{n} = \frac{n!}{n^{n}} $ είναι φραγμένη. 
        \hfill Απ: $ 0 \leq a_{n} \leq 1 $ 
    \item Να δείξετε ότι η ακολουθία $ a_{n} = \frac{\cos{n} + n \sin{n}}{n^{2}} $ 
        είναι φραγμένη. 
        \hfill Απ: $ \abs{an} \leq 2 $ 
    \item Να δείξετε ότι η ακολουθία $ a_{n} = \frac{5 \cos^{3}{n}}{n+2} $ είναι 
        φραγμένη.
        \hfill Απ: $ \abs{a_{n}} < \frac{5}{2}  $ 


\end{enumerate}
